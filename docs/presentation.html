HOW IT WORKS
5 lines. That's it.
From install to persistent memory in under 10 seconds.
from aura import Aura, Level

brain = Aura("./memory")

brain.store("User prefers dark mode",
         level=Level.Identity, tags=["ui"])

context = brain.recall("preferences") # <1ms

# inject into any LLM system prompt
